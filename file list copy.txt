import React, { useState } from "react";

interface UploadedFile {
  id: string;
  file: File;
  status: "Ready" | "Converting" | "completed" | "error";
  errorMessage?: string;
}

interface FileListProps {
  files: UploadedFile[];
  operationtype: "convert" | "merge" | "compress" | "split";
  title: string;
  onRemove?: (id: string) => void;
  onOperation?: (operation: string, files: UploadedFile[]) => void;
}

const operationDetails: Record<
  FileListProps["operationtype"],
  { description: string; buttonLabel: string }
> = {
  convert: {
    description:
      "Convert your files quickly and easily. Supported formats include PDF, Word, Excel, and more.",
    buttonLabel: "Convert PDF",
  },
  merge: {
    description: "Merge multiple PDF files into a single document seamlessly.",
    buttonLabel: "Merge PDF",
  },
  compress: {
    description:
      "Compress your PDF files to reduce their size without compromising quality.",
    buttonLabel: "Compress PDF",
  },
  split: {
    description: "Split your PDF files into smaller, manageable documents.",
    buttonLabel: "Split PDF",
  },
};

export default function FileList({
  files,
  operationtype,
  title,
  onRemove,
  onOperation,
}: FileListProps) {
  const [isProcessing, setIsProcessing] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // Simplified operation handler
  const handleOperation = () => {
    if (!onOperation) return;
    setIsProcessing(true);
    setError(null);
    try {
      onOperation(operationtype, files);
    } catch {
      setError("Operation failed, please try again.");
    } finally {
      setIsProcessing(false);
    }
  };

  return (
    <div className="flex flex-col md:flex-row h-full min-h-screen w-full">
      {/* LEFT: File List */}
      <section className="flex-1 overflow-y-auto p-6 flex flex-col items-center bg-gray-100">
        <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6 justify-center max-w-[800px] ">
          {files.length === 0 ? (
            <p className="text-gray-500 text-center col-span-full">
              No files uploaded yet.
            </p>
          ) : (
            files.map(({ id, file, status, errorMessage }) => (
              <div
                key={id}
                className="group relative w-auto h-auto mt-20 rounded-sm border border-gray-200 bg-white p-2 shadow-sm hover:shadow-md transition-shadow"
              >
                {/* Preview */}
                {file.type === "application/pdf" ? (
                  <embed
                    src={URL.createObjectURL(file)}
                    type="application/pdf"
                    className="w-full h-50 rounded mb-2 overflow-hidden"
                    onLoad={(e) =>
                      URL.revokeObjectURL((e.target as any).src)
                    }
                  />
                ) : file.type.startsWith("image/") ? (
                  <img
                    src={URL.createObjectURL(file)}
                    alt={file.name}
                    className="w-full h-40 object-contain rounded mb-2 overflow-hidden"
                    onLoad={(e) =>
                      URL.revokeObjectURL((e.target as HTMLImageElement).src)
                    }
                  />
                ) : (
                  <div className="flex items-center justify-center w-full h-40 bg-gray-100 rounded mb-2 text-gray-400 text-sm select-none">
                    No preview available
                  </div>
                )}

                {/* File info */}
                <p
                  className="text-sm  text-gray-900 truncate"
                  title={file.name}
                >
                  {file.name}
                </p>
                {status === "error" && errorMessage && (
                  <p
                    className="text-red-500 text-xs mt-1 truncate"
                    title={errorMessage}
                  >
                    {errorMessage}
                  </p>
                )}

                {/* Remove button - only if onRemove passed */}
                {onRemove && (
                  <button
                    onClick={() => onRemove(id)}
                    className="absolute top-3 right-3 opacity-0 group-hover:opacity-100
                               bg-red-600 hover:bg-red-700 text-white rounded-full
                               w-6 h-6 flex items-center justify-center
                               transition-opacity duration-300 focus:outline-none"
                    aria-label={`Remove file ${file.name}`}
                    type="button"
                  >
                    <svg
                      xmlns="http://www.w3.org/2000/svg"
                      className="w-4 h-4"
                      fill="none"
                      viewBox="0 0 24 24"
                      stroke="currentColor"
                      strokeWidth={2}
                    >
                      <path
                        strokeLinecap="round"
                        strokeLinejoin="round"
                        d="M6 18L18 6M6 6l12 12"
                      />
                    </svg>
                  </button>
                )}
              </div>
            ))
          )}
        </div>
      </section>

      {/* RIGHT: Operation Panel */}
      <aside className="w-full md:w-[420px] p-6 flex flex-col bg-white shadow-lg py-30">
        <h2 className="text-3xl font-bold mb-4">{title}</h2>

        <p className="text-gray-700 mb-6">
          {operationDetails[operationtype].description}
        </p>

        {error && (
          <p
            className="mb-4 text-red-600 font-semibold"
            role="alert"
            aria-live="assertive"
          >
            {error}
          </p>
        )}

        <button
          type="button"
          onClick={handleOperation}
          disabled={isProcessing}
          className={`w-full mt-6 py-4 rounded font-semibold cursor-pointer text-white transition-colors focus:outline-none ${
            isProcessing
              ? "bg-red-300 cursor-not-allowed"
              : "bg-red-600 hover:bg-red-700"
          }`}
          aria-busy={isProcessing}
        >
          {isProcessing
            ? `${operationDetails[operationtype].buttonLabel}...`
            : operationDetails[operationtype].buttonLabel}
        </button>
      </aside>
    </div>
  );
}
